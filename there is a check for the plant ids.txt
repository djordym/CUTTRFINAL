public async Task CreateMatchAsync(int plantId1, int plantId2, int userId1, int userId2)
{
    // Ensure PlantId1 is less than PlantId2
    if (plantId1 > plantId2)
    {
        // Swap PlantId1 and PlantId2
        int tempPlantId = plantId1;
        plantId1 = plantId2;
        plantId2 = tempPlantId;

        // Swap UserId1 and UserId2 correspondingly
        int tempUserId = userId1;
        userId1 = userId2;
        userId2 = tempUserId;
    }

    var match = new MatchEF
    {
        PlantId1 = plantId1,
        PlantId2 = plantId2,
        UserId1 = userId1,
        UserId2 = userId2,
        CreatedAt = DateTime.UtcNow
    };

    // Save the match to the database
    await _matchRepository.AddAsync(match);
}
